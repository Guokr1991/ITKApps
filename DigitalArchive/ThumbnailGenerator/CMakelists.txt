FIND_PACKAGE(ITK)

IF(ITK_FOUND)
  INCLUDE(${ITK_USE_FILE})
ELSE(ITK_FOUND)
  MESSAGE(FATAL_ERROR
          "Cannot build InsightApplications without ITK.  Please set ITK_DIR.")
 ENDIF(ITK_FOUND)

ADD_EXECUTABLE(GenerateThumbnail GenerateThumbnail.cxx)
TARGET_LINK_LIBRARIES(GenerateThumbnail ITKCommon ITKBasicFilters ITKIO)

ADD_EXECUTABLE(DicomGenerateThumbnail DicomGenerateThumbnail.cxx)
TARGET_LINK_LIBRARIES(DicomGenerateThumbnail ITKCommon ITKBasicFilters ITKIO)
