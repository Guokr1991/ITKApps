PROJECT(CLReg)

IF( BUILD_OUTSIDE_INSIGHT_APPLICATIONS )

  FIND_PACKAGE(ITK)
  IF(ITK_FOUND)
    INCLUDE(${ITK_USE_FILE})
  ELSE(ITK_FOUND)
    MESSAGE(FATAL_ERROR
            "Cannot build ITKApps without ITK.  Please set ITK_DIR.")
  ENDIF(ITK_FOUND)

ENDIF( BUILD_OUTSIDE_INSIGHT_APPLICATIONS )


INCLUDE_DIRECTORIES(
${ITK_SOURCE_DIR}/Code/Common
${ITK_SOURCE_DIR}/Code/IO
${ITK_SOURCE_DIR}/Code/BasicFilters
${ITK_SOURCE_DIR}/Code/Numerics
)

LINK_DIRECTORIES(
${ITK_BINARY_DIR}/Code/Common
${ITK_BINARY_DIR}/Code/IO
${ITK_BINARY_DIR}/Code/BasicFilters
${ITK_BINARY_DIR}/Code/Numerics
)

LINK_LIBRARIES( ITKCommon ITKBasicFilters ${ITK_LIBRARIES})
CONFIGURE_FILE(${CLReg_SOURCE_DIR}/testScript.in
                ${CLReg_BINARY_DIR}/testScript IMMEDIATE )
ADD_EXECUTABLE(CLReg CLReg.cxx)
INSTALL_TARGETS(/bin CLReg)
ADD_EXECUTABLE(CLRegTest CLRegTest.cxx)
INSTALL_TARGETS(/bin CLRegTest)
